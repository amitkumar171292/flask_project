# flask_project

# Prerequisites:

1. Flask: pip install flask
2. Flask-SQLAlchemy: pip install flask-sqlalchemy
3. Jinja2: pip install Jinja2

# Steps to creste tables in sqllite:
1. sqlite3 your_database.db
2. .read schema.sql
3. .exit


# Steps to view data

# list all the tables
.tables

# list schema of specific table
.schema users

# list all the data
SELECT * FROM table_name;

# How to Run?
python app.py (in terminal)
